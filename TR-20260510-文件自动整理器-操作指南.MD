### 1.2 最终命名：FileSort

**选择理由**：

1. **ASO搜索优化（最关键）**：
   - "File" + "Sort" 是App Store搜索量最高的文件管理关键词组合
   - Apple算法中App名称权重最高，"FileSort"同时覆盖"file"和"sort"两个高搜索量关键词
   - 用户搜索"file organizer"、"sort files"、"file sort"时均可命中
   - 30字符限制内：`FileSort: Auto File Organizer` = 30字符（完美利用）

2. **品牌记忆度（最最关键）**：
   - 两个单音节词组合：File + Sort = 8字母，极简
   - 发音节奏感强：/faɪl sɔːrt/，朗朗上口
   - 类似成功案例：FileZilla、Dropbox、Sortly — 短名+功能词组合
   - 用户口口相传时："Just use FileSort" — 一句话就能记住

3. **功能描述性（很关键）**：
   - 名字即功能：File + Sort = 文件排序/整理
   - 副标题"Auto File Organizer"补充自动化属性
   - 无需解释用户即知用途

4. **情感共鸣**：
   - "Sort"暗示秩序、整洁、掌控感
   - 对比竞品"Documents by Readdle"（太正式）或"FileBrowser"（太技术）
   - FileSort传递"一键搞定"的爽感

5. **国际化适配**：
   - "File"和"Sort"是几乎所有欧洲语言的通用词
   - 无文化歧义，全球通用

### 1.3 ASO关键词策略

### 6.1 数据流关键逻辑

**文件分类规则引擎（核心算法）**：

```
输入: ScannedFile + SortRule[]
输出: 目标文件夹路径

算法:
1. 获取文件的UTType (Uniform Type Identifier)
2. 按规则优先级从高到低匹配:
   a. 用户自定义规则 (priority: 100)
   b. 文件扩展名规则 (priority: 50)
   c. UTType类别规则 (priority: 30)
   d. 默认规则 (priority: 10)
3. 首个匹配的规则决定目标路径
4. 无匹配 → 移动到 /Other 文件夹

UTType分类映射:
  .pdf/.doc/.docx/.txt/.rtf/.pages → Documents
  .jpg/.png/.heic/.gif/.webp/.svg  → Images
  .mp4/.mov/.avi/.mkv/.m4v         → Videos
  .mp3/.m4a/.wav/.aac/.flac        → Music
  .zip/.rar/.7z/.tar/.gz           → Archives
  .xls/.xlsx/.csv/.numbers         → Spreadsheets
  .ppt/.pptx/.key                  → Presentations
  .xcodeproj/.swift/.py/.js        → Code
  其他                              → Other
```

**冲突解决策略**：

```
当目标路径已存在同名文件时:
  1. SHA256哈希比对
     ├── 哈希相同 → 标记为重复，跳过（保留较新版本）
     └── 哈希不同 → 执行冲突解决策略:
         a. 添加序号: "report.pdf" → "report_2.pdf"
         b. 添加日期: "report.pdf" → "report_2026-05-10.pdf"
         c. 用户选择: 保留两者/替换/跳过
```

### 7.3 规则引擎

```swift
import Foundation

struct SortAction: Identifiable {
    let id = UUID()
    let sourceURL: URL
    let destinationURL: URL
    let category: FileCategory
    let matchedRule: SortRule?
    let isConflict: Bool
    let conflictType: ConflictType?

    enum ConflictType {
        case nameCollision
        case duplicateContent
    }
}

@MainActor
final class RuleEngine: ObservableObject {
    private var rules: [SortRule] = []
    private let defaultCategories = FileCategory.allCases.filter { $0 != .other }

    func loadRules(modelContext: ModelContext) {
        let descriptor = FetchDescriptor<SortRule>(sortBy: [SortDescriptor(\.priority, order: .reverse)])
        rules = (try? modelContext.fetch(descriptor)) ?? []
    }

    func resolve(files: [ScannedFile], baseDirectory: URL) -> [SortAction] {
        var actions: [SortAction] = []
        let activeRules = rules.filter(\.isEnabled)

        for file in files {
            let (destURL, matchedRule) = resolveDestination(
                file: file,
                rules: activeRules,
                baseDirectory: baseDirectory
            )

            let isConflict = FileManager.default.fileExists(atPath: destURL.path)
            let conflictType: SortAction.ConflictType? = isConflict ? .nameCollision : nil

            actions.append(SortAction(
                sourceURL: file.url,
                destinationURL: destURL,
                category: file.category,
                matchedRule: matchedRule,
                isConflict: isConflict,
                conflictType: conflictType
            ))
        }
        return actions
    }

    private func resolveDestination(file: ScannedFile, rules: [SortRule], baseDirectory: URL) -> (URL, SortRule?) {
        for rule in rules where matchesRule(file: file, rule: rule) {
            let dest = baseDirectory.appendingPathComponent(rule.destinationPath).appendingPathComponent(file.url.lastPathComponent)
            return (dest, rule)
        }

        let categoryPath = file.category.rawValue
        let dest = baseDirectory.appendingPathComponent(categoryPath).appendingPathComponent(file.url.lastPathComponent)
        return (dest, nil)
    }

    private func matchesRule(file: ScannedFile, rule: SortRule) -> Bool {
        switch rule.conditionType {
        case .fileExtension:
            return file.fileExtension == rule.conditionValue.lowercased()
        case .utTypeCategory:
            return file.category.rawValue == rule.conditionValue
        case .fileNameContains:
            return file.name.localizedCaseInsensitiveContains(rule.conditionValue)
        case .fileSizeGreaterThan:
            guard let threshold = Int64(rule.conditionValue) else { return false }
            return file.size > threshold
        case .fileSizeLessThan:
            guard let threshold = Int64(rule.conditionValue) else { return false }
            return file.size < threshold
        case .createdAfter, .createdBefore, .modifiedAfter:
            guard let date = DateFormatter.iso8601.date(from: rule.conditionValue) else { return false }
            if rule.conditionType == .createdAfter { return file.createdAt > date }
            if rule.conditionType == .createdBefore { return file.createdAt < date }
            return file.modifiedAt > date
        }
    }
}

extension DateFormatter {
    static let iso8601: DateFormatter = {
        let f = DateFormatter()
        f.dateFormat = "yyyy-MM-dd"
        return f
    }()
}
```

### 7.4 重复文件检测器

```swift
import CryptoKit
import Foundation

struct DuplicateGroup: Identifiable {
    let id = UUID()
    let hash: String
    let files: [URL]
    let totalSize: Int64
}

final class DuplicateDetector {
    static func findDuplicates(in actions: [SortAction]) -> [DuplicateGroup] {
        var hashMap: [String: [URL]] = [:]

        for action in actions where action.isConflict {
            let hash = computeHash(for: action.sourceURL)
            if let existing = hashMap[hash] {
                hashMap[hash] = existing + [action.sourceURL]
            } else {
                hashMap[hash] = [action.sourceURL, action.destinationURL]
            }
        }

        return hashMap.compactMap { hash, urls in
            guard urls.count > 1 else { return nil }
            let size = (try? FileManager.default.attributesOfItem(atPath: urls[0].path)[.size] as? Int64) ?? 0
            return DuplicateGroup(hash: hash, files: urls, totalSize: size * Int64(urls.count - 1))
        }
    }

    static func computeHash(for url: URL) -> String {
        guard let data = try? Data(contentsOf: url, options: .alwaysMapped) else { return UUID().uuidString }
        let sha256 = SHA256.hash(data: data)
        return sha256.compactMap { String(format: "%02x", $0) }.joined()
    }
}
```

### 7.5 文件执行器（含撤销）

```swift
import Foundation

struct UndoSnapshot: Codable {
    let id: UUID
    let timestamp: Date
    let operations: [FileOperation]
}

struct FileOperation: Codable {
    let source: String
    let destination: String
    let operationType: OperationType

    enum OperationType: String, Codable {
        case move
        case copy
    }
}

@MainActor
final class FileExecutor: ObservableObject {
    @Published var isExecuting = false
    @Published var progress: Double = 0
    @Published var completedCount = 0

    private var undoStack: [UndoSnapshot] = []
    private let maxUndoHistory = 50

    func execute(actions: [SortAction], mode: FileOperation.OperationType = .move) async throws -> Int {
        isExecuting = true
        completedCount = 0
        defer { isExecuting = false }

        var operations: [FileOperation] = []
        let fileManager = FileManager.default

        for (index, action) in actions.enumerated() {
            let destDir = action.destinationURL.deletingLastPathComponent()
            if !fileManager.fileExists(atPath: destDir.path) {
                try fileManager.createDirectory(at: destDir, withIntermediateDirectories: true)
            }

            let finalDest = resolveNameConflict(action.destinationURL)

            switch mode {
            case .move:
                try fileManager.moveItem(at: action.sourceURL, to: finalDest)
            case .copy:
                try fileManager.copyItem(at: action.sourceURL, to: finalDest)
            }

            operations.append(FileOperation(
                source: action.sourceURL.path,
                destination: finalDest.path,
                operationType: mode
            ))

            completedCount = index + 1
            progress = Double(completedCount) / Double(actions.count)
        }

        let snapshot = UndoSnapshot(id: UUID(), timestamp: Date(), operations: operations)
        undoStack.append(snapshot)
        if undoStack.count > maxUndoHistory { undoStack.removeFirst() }

        return completedCount
    }

    func undo() async throws {
        guard let snapshot = undoStack.popLast() else { return }
        let fileManager = FileManager.default

        for op in snapshot.operations.reversed() {
            let source = URL(fileURLWithPath: op.source)
            let dest = URL(fileURLWithPath: op.destination)
            switch op.operationType {
            case .move:
                try fileManager.moveItem(at: dest, to: source)
            case .copy:
                try fileManager.removeItem(at: dest)
            }
        }
    }

    private func resolveNameConflict(_ url: URL) -> URL {
        let fileManager = FileManager.default
        guard fileManager.fileExists(atPath: url.path) else { return url }

        let name = url.deletingPathExtension().lastPathComponent
        let ext = url.pathExtension
        var counter = 2

        while true {
            let newName = ext.isEmpty ? "\(name)_\(counter)" : "\(name)_\(counter).\(ext)"
            let newURL = url.deletingLastPathComponent().appendingPathComponent(newName)
            if !fileManager.fileExists(atPath: newURL.path) { return newURL }
            counter += 1
        }
    }
}
```

### 7.6 编写规则

1. **Swift并发**：所有FileManager操作使用`@MainActor`标注，异步操作使用`async/await`
2. **SwiftData**：数据模型使用`@Model`宏，查询使用`@Query`属性包装器
3. **UTType**：文件类型判断统一使用`UniformTypeIdentifiers`框架，不硬编码扩展名
4. **错误处理**：所有文件操作使用`try` + 自定义`FileSortError`枚举
5. **撤销安全**：每次操作前保存快照，撤销操作反向执行
6. **沙盒合规**：使用`Security-Scoped Bookmarks`持久化用户授权的目录访问
7. **内存管理**：大文件哈希使用流式读取（分块SHA256），不一次性加载到内存
